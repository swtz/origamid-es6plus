// manipulação dos valores de entrada

// tipo 'checkbox' aceita mais de um valor
// tipo 'radio' aceita um valor por vez, entretanto é preciso
// agrupá-los por passar o mesmo nome no attr 'name' deles.
// Se o attr 'value' estiver omitido, retorna 'on'.
// Ambos, quando selecionados, possuem o attr 'checked'.

// Dica para capturar todos os dados de um formulário e atribuir os
// valores a um objeto vazio:
const formObject = {};
function handleChange(event) {
  formObject[event.target.name] = event.target.value;
}
form.addEventListener('change', handleChange);
