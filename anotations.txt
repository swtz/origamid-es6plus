// AULA 0502 | MODULES

export default function sayHello() {
  return 'Hello!';
}

// export default -> import anyNameForFn from './path/script.js'; 
// É usado para exportar SOMENTE 1 variável
// Não é necessário importar a função com o mesmo nome

// #!important
// Ao reunir a execução dos módulos em um arquivo só, por exemplo, no arquivo 'script.js'
// Ao usá-lo no HTML por meio da tag <script></script>
// Somente vai funcionar passando o attr 'type="module"'


// into file script.js
export function sayHelloWorld() {
  return 'Hello, world!';
}

export function sayHelloDear() {
  return 'Hello, dear!';
}

// Named Export
// export -> import { sayHelloWorld, sayHelloDear } from './path/script.js';
// É usado para exportar MAIS DE 1 variável
// É necessário importar a função com o mesmo nome
// É possível 'apelidar' a variável -> import { sayHelloWorld as sayHW } from './path/script.js';
// Caso o dev não saiba dos métodos que o módulo possui -> import * as greetings from './path/script.js';
// Uso: greetings.sayHelloDear();


// Normalmente, o módulo que é exportado com 'export default' indica que o módulo
// realiza alguma função específica
