const observer = new MutationObserver(handleMutation);
// recebe uma callbackFn no arg0

observer.observe(observeTarget);
// recebe um objeto do tipo 'Node' no arg0

observer.observe(observeTarget, { attributes: true });
// recebe um 'config-object' no arg1
// nesse caso, observará alguma mutação num dos atributos
// do 'observeTarget'

function callbackFn(mutation) {
  console.log(mutation);
}
// a callbackFn possui um param especial
// retorna um objeto do tipo 'Mutation Record'
// que contém todas as mutações ocorridas no 'observeTarget'
// Nesse exemplo, observará apenas as mudanças nos attr's
