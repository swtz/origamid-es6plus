// Tip for mobile: eventListener 'touchstart' dispara o evento de "click" na hora
// caso o dev use 'click', no ambiente mobile, o evento de "click" demora 300ms
// para ocorrer

// para add vários eventos em 1 (ou mais) elementos:
NodeList.forEach((HTMLElement) => {
  ['touchstart', 'click'].forEach(userEvent => {
    HTMLElement.addEventListener(userEvent, handleClick)
  })
});

// depois, basta desenvolver a fn handleClick
