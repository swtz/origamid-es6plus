// No javascript, é possível ter 'setters' sem os 'getters'.
// Porém, não é possível ter 'getters' sem os 'setters'.
class Fruteira {
  constructor(listaFrutas) {
    this.lista = listaFrutas || [];
  }

  set nova(fruta) {
    this.lista.push(fruta);
  }
}
const frutas = new Fruteira();
frutas.nova = 'Pêra'
frutas.nova = 'Uva'
frutas.nova = 'Laranja'
frutas.nova = 'Banana'


// Abaixo, um 'setter' modificando o comportamento de um 'getter'.
class Button {
  constructor(text, color) {
    this.text = text;
    this.color = color;
  }

  get element() {
    const type = this._elementType || 'button';
    const btnElement = document.createElement(type);
    btnElement.innerText = this.text;
    btnElement.style.color = this.color;
    return btnElement;
  }

  set element(value) {
    this._elementType = value;
  }
}
const blueBtn = new Button('Comprar', 'blue');
blueBtn.element = 'a'; // retorna um elemento 'a' ao invés de 'button'.
