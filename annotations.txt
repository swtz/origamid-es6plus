// Todo tipo iterável deve conter esse método 
// (geralmente está no protótipo do tipo).
Symbol(Symbol.iterator);
// Esse método é quem sabe iterar sobre o tipo do dado,
// ou seja, cada iterável possui uma implementação diferente.


// Uma outra maneira de se fazer loop em uma 'array-like', por exemplo
// Essa instrução somente funcionará objetos do tipo 'Iterable'
// key word 'for' junto com 'of'.
const btns = document.querySelectorAll('button');
for (const btn of btns) {
  btn.style.backgroundColor = 'aqua';
}

// Na verdade, é isso que o '...' faz por debaixo dos panos.
// Por isso, o 'spread' só pode ser usado com objetos do tipo
// 'Iterable'


// Para iterar sobre um dado que não contém
Symbol(Symbol.iterator);
// ou seja, que não é 'Iterable':
// key word 'for' junto com 'in'.
const pessoa = {
  nome: 'Leonard',
  idade: 25
}
for (const key in pessoa) {
  console.log(pessoa[key])
  // 'key' retorna as chaves ENUMERÁVEIS do objeto
  // e que não sejam símbolos.
}


// loop: do / while
let i = 0;
do {
  console.log(i++)
} while (i < 10);
