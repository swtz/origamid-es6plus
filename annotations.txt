// 'Polyfill'
// Cria um método com o mesmo nome do recurso moderno, porém,
// internamente, usa código antigo para compatibilizar com
// browsers antigos. O código interno é denominado 'polyfill'.

// Transpiler
// Muda a sintaxe. Transforma código novo em código antigo.

// O 'Babel' faz os 2 processos.

// Exemplo de um 'polyfill' do método 'fetch':
// https://github.com/JakeChampion/fetch

// [webpack.config.js]:
const path = require('path');
module.exports = {
  mode: 'development',
  entry: ['@babel/polyfill', 'whatwg-fetch', './js/script.js'],
  output: {
    path: path.resolve(__dirname, './'),
    filename: 'main.js',
  },
};
// 'whatwg-fetch' é o nome do polyfill
