'$&'; // faz referência ao que foi selecionado na expressão do regex.

/(\w+)@([\w.]+)/gi; // os '()' criam um 'grupo de captura';
// é util para isolar valores da seleção com base na quantidade
// de instruções 'regex'.
// No exemplo, há 2 'grupos de captura'.
// Podem ser acessados pelo caractere '$n', ou seja,
// $1 -> primeiro grupo
// $2 -> segundo grupo


/(ta)+/gi; // os '()' concatenam uma sequência, ou seja,
// será procurado uma sequência exata dos caracteres 'ta'.
// Também é criado um 'grupo de captura'.
// Caso não se queira criar o grupo de captura:
/(?:ta)+/gi; // seleciona da mesma maneira, porém nenhum grupo é criado.


// Positive Lookahead
/\d(?=px)/gi; // Nesse exemplo, procura todos os dígitos que contenham
// os caracteres 'px' à frente deles.

// Negative Lookahead
/\d(?!px)/gi; // Nega a seleção acima, ou seja, todo digito que não houver
// 'px' à frente será selecionado.
